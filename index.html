import os import shutil def organize_folder(target_folder): # Define categories and their corresponding file extensions FILE_CATEGORIES = { "Documents": [".pdf", ".docx", ".doc", ".txt", ".xlsx", ".pptx", ".csv"], "Images": [".jpg", ".jpeg", ".png", ".gif", ".bmp", ".svg", ".webp"], "Videos": [".mp4", ".mkv", ".avi", ".mov", ".flv", ".wmv"], "Audio": [".mp3", ".wav", ".aac", ".flac", ".ogg", ".m4a"], "Archives": [".zip", ".rar", ".7z", ".tar", ".gz"], "Installers": [".exe", ".msi", ".dmg", ".pkg"] } # Verify if the provided folder path exists if not os.path.exists(target_folder): print(f"Error: The path '{target_folder}' does not exist.") return print(f"Organizing files in: {target_folder}...") moved_count = 0 # Iterate through all items in the target directory for item in os.listdir(target_folder): item_path = os.path.join(target_folder, item) # Skip directories to only process files if os.path.isdir(item_path): continue # Extract file extension in lowercase _, file_extension = os.path.splitext(item) file_extension = file_extension.lower() # Identify the correct folder category for the file target_category = "Others" for category, extensions in FILE_CATEGORIES.items(): if file_extension in extensions: target_category = category break # Construct the destination folder path category_folder_path = os.path.join(target_folder, target_category) # Create the subfolder if it does not already exist if not os.path.exists(category_folder_path): os.makedirs(category_folder_path) # Move the file to its designated category folder try: shutil.move(item_path, os.path.join(category_folder_path, item)) moved_count += 1 except Exception as e: print(f"Could not move {item}: {e}") print(f"Success! {moved_count} files successfully organized.") # Example Usage: Replace with your actual directory path # Use a double backslash '\\' or a forward slash '/' for paths on Windows folder_to_clean = "./my_cluttered_folder" organize_folder(folder_to_clean)
